Updated Props Design
User ke feedback ke baad:

interface AvtPr {
  src?:       string;           // both modes — image URL
  name?:      string;           // initials source + alt default
  alt?:       string;           // overrides name-based alt
  shape?:     'ic' | 'sq';     // default: 'ic'
  // Normal mode (default, no flag needed):
  checked?:   boolean;          // selection state
  onSelect?:  () => void;       // selection callback
  // Upload mode (uploadable flag activates it):
  uploadable?: boolean;         // switches to upload mode — unmounts checkbox entirely
  onUpload?:  (file: File) => void;
}

JSX after implementation:

// Normal — image + selection

<Avt src={image} name={name} shape="ic" uploadable onUpload={handleFile} />

<Avt name={name} shape="ic" uploadable onUpload={handleFile} />
